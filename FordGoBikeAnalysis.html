#function so we can make age group based on member_birth_year column
def age (x):
   # age = this year minus his birth day
    age = 2019-x
    if age < 18:
        #if he less than 18 he is teenager 
        return 'Teen'
    if  age <= 35 and age >= 18 :
        #between 35 and 18 he is adult
        return 'Adult'
    elif age <= 55 and age >=36 :
        #between 55 and 36 he is middle aged
        return 'Middle-Aged Adults'
    else:
        # > 55 he is senior 
        return 'Senior'
